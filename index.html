import "jsr:@supabase/functions-js/edge-runtime.d.ts"; import { createClient } from "jsr:@supabase/supabase-js@2"; Deno.serve(async (req) => { try { const supabaseUrl = Deno.env.get("SUPABASE_URL")!; const supabaseKey = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!; const supabase = createClient(supabaseUrl, supabaseKey); const TELEGRAM_BOT_TOKEN = Deno.env.get("TELEGRAM_BOT_TOKEN"); const TELEGRAM_CHAT_ID = Deno.env.get("TELEGRAM_CHAT_ID"); const now = new Date(); const romeTime = new Date(now.toLocaleString("en-US", { timeZone: "Europe/Rome" })); const currentTotalMinutes = romeTime.getHours() * 60 + romeTime.getMinutes(); const inizioOggi = new Date(romeTime); inizioOggi.setHours(0, 0, 0, 0); const fineOggi = new Date(romeTime); fineOggi.setHours(23, 59, 59, 999); // data di oggi in formato YYYY-MM-DD (fuso orario Italia) const oggiStr = `${romeTime.getFullYear()}-${String(romeTime.getMonth() + 1).padStart(2, "0")}-${String(romeTime.getDate()).padStart(2, "0")}`; console.log(`[INFO] Controllo terapie personali. Ora italiana: ${romeTime.toLocaleTimeString()} (Minuti: ${currentTotalMinutes})`); const { data: orariTerapie, error: errorOrari } = await supabase .from("terapie_orari") .select(` id, ora, quantita, terapia_id, terapie ( id, nome, dosaggio, unita_dosaggio, attiva ) `); if (errorOrari) { console.error("[ERRORE DATABASE ORARI]", errorOrari); throw errorOrari; } // posticipi impostati per oggi dall'app const { data: posticipiOggi, error: errorPosticipi } = await supabase .from("terapie_posticipi") .select("terapia_id, orario_id, nuovo_orario") .eq("data", oggiStr); if (errorPosticipi) { console.error("[ERRORE DATABASE POSTICIPI]", errorPosticipi); } const posticipiMap = new Map( (posticipiOggi || []).map((p) => [`${p.terapia_id}_${p.orario_id}`, p.nuovo_orario]) ); const candidati = (orariTerapie || []).filter((item) => { if (!item.terapie || item.terapie.attiva !== true || !item.ora) { return false; } const chiavePosticipo = `${item.terapia_id}_${item.id}`; const posticipo = posticipiMap.get(chiavePosticipo); let orarioMinuti; if (posticipo) { const dPost = new Date(new Date(posticipo).toLocaleString("en-US", { timeZone: "Europe/Rome" })); orarioMinuti = dPost.getHours() * 60 + dPost.getMinutes(); } else { const [oreStr, minStr] = item.ora.split(":"); orarioMinuti = parseInt(oreStr) * 60 + parseInt(minStr); } const differenza = Math.abs(orarioMinuti - currentTotalMinutes); return differenza <= 7; }); console.log(`[INFO] Terapie nell'intervallo orario corrente: ${candidati.length}`); for (const item of candidati) { const t = item.terapie; // Controlla se è già stata presa OGGI a QUESTO orario specifico // (prima controllava solo la terapia, saltando per errore anche gli altri orari dello stesso giorno) const { data: assunzioniOggi, error: errorAssunzioni } = await supabase .from("terapie_assunzioni") .select("id, assunta") .eq("terapia_id", item.terapia_id) .eq("orario_id", item.id) .eq("assunta", true) .gte("data_ora", inizioOggi.toISOString()) .lte("data_ora", fineOggi.toISOString()); if (errorAssunzioni) { console.error(`[ERRORE CONTROLLO ASSUNZIONI] Terapia ${t.nome}:`, errorAssunzioni); continue; } if (assunzioniOggi && assunzioniOggi.length > 0) { console.log(`[INFO] Salto invio per "${t.nome}": risulta già assunta oggi a quest'orario.`); continue; } const dosaggioCompleto = t.dosaggio ? `${t.dosaggio} ${t.unita_dosaggio || ""}`.trim() : ""; const quantitaDaPrendere = item.quantita ? `(Q.tà: ${item.quantita})` : ""; const chiavePosticipo = `${item.terapia_id}_${item.id}`; const eraPosticipata = posticipiMap.has(chiavePosticipo); const orarioMostrato = eraPosticipata ? new Date(posticipiMap.get(chiavePosticipo)).toLocaleTimeString("it-IT", { timeZone: "Europe/Rome", hour: "2-digit", minute: "2-digit", }) : item.ora.substring(0, 5); const testo = `🩺 **Promemoria Terapia**\n\nÈ il momento di assumere:\n💊 *${t.nome}* ${dosaggioCompleto} ${quantitaDaPrendere}\n⏰ Ore: ${orarioMostrato}${eraPosticipata ? " (posticipata)" : ""}`; console.log(`[TELEGRAM] Invio promemoria per: ${t.nome}`); const res = await fetch(`https://api.telegram.org/bot${TELEGRAM_BOT_TOKEN}/sendMessage`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ chat_id: TELEGRAM_CHAT_ID, text: testo, parse_mode: "Markdown", }), }); if (!res.ok) { const errText = await res.text(); console.error(`[ERRORE TELEGRAM] Fallito invio per ${t.nome}:`, errText); } else { console.log(`[TELEGRAM] Messaggio inviato con successo per ${t.nome}!`); } } return new Response(JSON.stringify({ status: "completato" }), { headers: { "Content-Type": "application/json" }, }); } catch (error) { console.error("[ERRORE GENERALE]", error.message); return new Response(JSON.stringify({ error: error.message }), { status: 500, headers: { "Content-Type": "application/json" }, }); } });
